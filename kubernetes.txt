安装docker https://www.runoob.com/docker/centos-docker-install.html

cd /home

curl -sSL https://get.daocloud.io/docker | sh

docker --version

设置仓库

sudo yum-config-manager --add-repo http://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo

sudo yum install docker-ce docker-ce-cli containerd.io

启动docker

sudo systemctl start docker



组件：

1、kube-proxy 负责管理集群内网络的

2、dashboard - web UI 管理集群的站点

3、kube-scheduler  调度器 - 依据策略节点资源调度容器与节点部署

4、kube-apiserver 通过api管理集群的入口


学习如何安装：

https://kubernetes.io/docs/tasks/tools/install-kubectl-linux/



下载：

cd /home 

curl -LO "https://dl.k8s.io/release/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl"

验证

curl -LO "https://dl.k8s.io/$(curl -L -s https://dl.k8s.io/release/stable.txt)/bin/linux/amd64/kubectl.sha256"

echo "$(cat kubectl.sha256)  kubectl" | sha256sum --check

安装：

sudo install -o root -g root -m 0755 kubectl /usr/local/bin/kubectl

查看版本

kubectl version --client





搭建集群的工具 https://minikube.sigs.k8s.io/docs/start/

安装

cat /proc/version

cd /home

curl -LO https://storage.googleapis.com/minikube/releases/latest/minikube-linux-amd64

sudo install minikube-linux-amd64 /usr/local/bin/minikube

# 启动集群
minikube start --force --driver=docker

kubectl get po -A

kubectl create deployment hello-minikube --image=k8s.gcr.io/echoserver:1.4

kubectl expose deployment hello-minikube --type=NodePort --port=8080

kubectl get services hello-minikube

minikube service hello-minikube

http://localhost:7080/



# 创建Nginx

kubectl create deployment nginx --image=nginx:1.14-alpine

kubectl get deploy

kubectl expose deploy nginx --port=80 --target-port=80 --type=NodePort
